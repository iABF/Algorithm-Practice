#include <bits/stdc++.h>

#define INF 0x7f7f7f7f
#define MAXN 47000
#define MOD 1000000007
#define ll long long
using namespace std;

ll n, a0, a1, b0, b1, i, A, B, C;

int gcd(ll m, ll e) {
    if (m == 0)return e;
    return gcd(e % m, m);
}

bool check(ll u, ll v) {
    return (gcd(u, A) == 1) && (gcd(B, v) == 1);
}

int main() {
    cin >> n;
    while (n--) {
        cin >> a0 >> a1 >> b0 >> b1;
        A = a0 / a1, B = b1 / b0, C = b1 / a1;
        ll ans = 0;
        for (i = 1; i * i < C; ++i) {
            if (C % i == 0) {
                ans += check(i, C / i);
                ans += check(C / i, i);
            }
        }
        if (C == i * i)ans += check(i, C / i);
        cout << ans << endl;
    }
}
